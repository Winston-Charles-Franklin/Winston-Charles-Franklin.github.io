import gmpy2
from Crypto.PublicKey import RSA
from Crypto.Util.number import bytes_to_long, long_to_bytes

with open("publickey1.pem","rb") as fp1:
    pk1bits = fp1.read()

with open("publickey2.pem","rb") as fp2:
    pk2bits = fp2.read()

with open("cipher1.txt","rb") as fpc1:
    c1 = fpc1.read()

with open("cipher2.txt","rb") as fpc2:
    c2 = fpc2.read()

c1 = bytes_to_long(c1)
c2 = bytes_to_long(c2)

pk1 = RSA.importKey(pk1bits)
pk2 = RSA.importKey(pk2bits)

n1 = pk1.n
n2 = pk2.n
e1 = pk1.e
e2 = pk2.e

print("n1:",n1)
print("n2:",n2)
print("e1:",e1)
print("e2:",e2)

n = n1

s = gmpy2.gcdext(e1,e2)

m = pow(c1,s[1],n)*pow(c2,s[2],n)%n

print(long_to_bytes(m).decode())
